INSTRUCCIONES PARA IA DEL FRONTEND - IMPLEMENTACION DE REPORTES DEL SISTEMA

IMPORTANTE - CONFIGURACION DEL INTERCEPTOR HTTP:
Si tu aplicacion frontend tiene un interceptor HTTP que requiere empresaId en todas las peticiones,
DEBES EXCLUIR la ruta '/api/reportes-sistema' de esa validacion, porque este endpoint NO requiere
empresaId (es un endpoint de sistema global, no filtrado por empresa).

Ejemplo de como excluir en el interceptor:
```typescript
// En tu HTTP interceptor
const rutasExcluidas = ['/api/reportes-sistema'];
if (rutasExcluidas.some(ruta => req.url.includes(ruta))) {
  return next.handle(req); // Pasar sin agregar empresaId
}
```

- POST /api/reportes-sistema/ejecutar/backup (ejecutar backup manualmente)

EJECUCION MANUAL DE REPORTES:
Ademas de la generacion automatica semanal, el sistema permite ejecutar reportes manualmente bajo demanda:

Ejecutar auditoria:
POST http://localhost:8080/api/reportes-sistema/ejecutar/auditoria
Respuesta: "Auditoria ejecutada correctamente. Refresque la lista para ver el nuevo reporte."

Ejecutar backup:
POST http://localhost:8080/api/reportes-sistema/ejecutar/backup
Respuesta: "Backup ejecutado correctamente. Refresque la lista para ver el nuevo archivo."

CRITICO - CONFIGURACION DE TIMEOUT:
Los scripts de auditoria y backup pueden tardar entre 10-30 segundos en completarse.
DEBES configurar el timeout del cliente HTTP a minimo 60 segundos.

Si usas fetch: Usa AbortController con timeout de 60000ms (ver codigo ejemplo)
Si usas axios: Configura timeout: 60000 en la peticion
Si usas tu propio cliente API: Asegurar timeout >= 60000ms

Ejemplos:

Con axios NO enviar body:
axios.post('/api/reportes-sistema/ejecutar/auditoria', null, { timeout: 60000 })

Con fetch NO enviar body:
fetch('/api/reportes-sistema/ejecutar/auditoria', { method: 'POST' })

CRITICO: NO envies ningun objeto en el body (ni siquiera {}). Los endpoints POST de ejecucion
no esperan parametros. Si envias un body vacio {}, causara errores.

IMPORTANTE: Durante la ejecucion, mostrar un spinner/loading y deshabilitar los botones
para evitar multiples ejecuciones simultaneas.

CODIGO DE EJEMPLO (React/TypeScript):

// Tipos
interface ReporteArchivo {
  nombre: string;
  tipo: 'AUDITORIA' | 'BACKUP';
  tamanoBytes: number;
  tamanoLegible: string;
  fechaCreacion: string;
  rutaRelativa: string;
}

interface ReportesResponse {
  auditorias: ReporteArchivo[];
  backups: ReporteArchivo[];
  totalAuditorias: number;
  totalBackups: number;
}

// Componente Principal
const ControladoresSistema = () => {
  const [reportes, setReportes] = useState<ReportesResponse | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const [vistaActual, setVistaActual] = useState<'auditorias' | 'backups'>('auditorias');
  const [ejecutando, setEjecutando] = useState<'auditoria' | 'backup' | null>(null);

  useEffect(() => {
    cargarReportes();
  }, []);

  const cargarReportes = async () => {
    setLoading(true);
    setError(null);
    try {
      const response = await fetch('http://localhost:8080/api/reportes-sistema');
      if (!response.ok) throw new Error('Error al cargar reportes');
      const data = await response.json();
      setReportes(data);
    } catch (err) {
      setError('No se pudieron cargar los reportes. Intente nuevamente.');
      console.error('Error:', err);
    } finally {
      setLoading(false);
    }
  };

  const ejecutarAuditoria = async () => {
    setEjecutando('auditoria');
    try {
      // IMPORTANTE: Los scripts pueden tardar 10-30 segundos
      // Asegurar que el timeout del cliente HTTP sea suficiente (60+ segundos)
      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), 60000); // 60 segundos
      
      const response = await fetch('http://localhost:8080/api/reportes-sistema/ejecutar/auditoria', {
        method: 'POST',
        signal: controller.signal
      });
      
      clearTimeout(timeoutId);
      
      if (!response.ok) throw new Error('Error al ejecutar auditoria');
      const mensaje = await response.text();
      alert(mensaje);
      // Recargar lista despues de 2 segundos
      setTimeout(() => cargarReportes(), 2000);
    } catch (err) {
      if (err.name === 'AbortError') {
        alert('La operacion tomo demasiado tiempo. Verifique los reportes manualmente.');
      } else {
        alert('Error al ejecutar auditoria: ' + err.message);
      }
      console.error('Error:', err);
    } finally {
      setEjecutando(null);
    }
  };

  const ejecutarBackup = async () => {
    setEjecutando('backup');
    try {
      // IMPORTANTE: Los scripts pueden tardar 10-30 segundos
      // Asegurar que el timeout del cliente HTTP sea suficiente (60+ segundos)
      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), 60000); // 60 segundos
      
      const response = await fetch('http://localhost:8080/api/reportes-sistema/ejecutar/backup', {
        method: 'POST',
        signal: controller.signal
      });
      
      clearTimeout(timeoutId);
      
      if (!response.ok) throw new Error('Error al ejecutar backup');
      const mensaje = await response.text();
      alert(mensaje);
      // Recargar lista despues de 2 segundos
      setTimeout(() => cargarReportes(), 2000);
    } catch (err) {
      if (err.name === 'AbortError') {
        alert('La operacion tomo demasiado tiempo. Verifique los reportes manualmente.');
      } else {
        alert('Error al ejecutar backup: ' + err.message);
      }
      console.error('Error:', err);
    } finally {
      setEjecutando(null);
    }
  };

  const descargarArchivo = (tipo: 'AUDITORIA' | 'BACKUP', nombreArchivo: string) => {
    const url = `http://localhost:8080/api/reportes-sistema/descargar/${tipo}/${nombreArchivo}`;
    window.open(url, '_blank');
  };

  const verAuditoria = (nombreArchivo: string) => {
    const url = `http://localhost:8080/api/reportes-sistema/ver/auditoria/${nombreArchivo}`;
    window.open(url, '_blank');
  };

  const formatearFecha = (fechaISO: string): string => {
    const fecha = new Date(fechaISO);
    return fecha.toLocaleString('es-AR', {
      year: 'numeric',
      month: 'long',
      day: 'numeric',
      hour: '2-digit',
      minute: '2-digit'
    });
  };

  if (loading) {
    return (
      <div className="loading-container">
        <div className="spinner"></div>
        <p>Cargando reportes del sistema...</p>
      </div>
    );
  }

  if (error) {
    return (
      <div className="error-container">
        <p>{error}</p>
        <button onClick={cargarReportes}>Reintentar</button>
      </div>
    );
  }

  return (
    <div className="controladores-sistema">
      <header>
        <div>
          <h1>Controladores del Sistema</h1>
          <p>Monitoreo y respaldo automatico</p>
        </div>
        <div className="acciones-header">
          <button 
            onClick={ejecutarAuditoria} 
            disabled={ejecutando !== null}
            className="btn-auditoria"
          >
            {ejecutando === 'auditoria' ? 'Ejecutando...' : 'Ejecutar Auditoria Ahora'}
          </button>
          <button 
            onClick={ejecutarBackup} 
            disabled={ejecutando !== null}
            className="btn-backup"
          >
            {ejecutando === 'backup' ? 'Ejecutando...' : 'Ejecutar Backup Ahora'}
          </button>
          <button onClick={cargarReportes} disabled={ejecutando !== null}>
            Actualizar Lista
          </button>
        </div>
      </header>

      {/* Tarjetas de Resumen */}
      <div className="resumen-cards">
        <div className="card">
          <h3>Auditorias Disponibles</h3>
          <p className="numero">{reportes?.totalAuditorias || 0}</p>
          <button onClick={() => setVistaActual('auditorias')}>Ver Detalle</button>
        </div>
        <div className="card">
          <h3>Backups Disponibles</h3>
          <p className="numero">{reportes?.totalBackups || 0}</p>
          <button onClick={() => setVistaActual('backups')}>Ver Detalle</button>
        </div>
      </div>

      {/* Pestañas */}
      <div className="tabs">
        <button 
          className={vistaActual === 'auditorias' ? 'active' : ''}
          onClick={() => setVistaActual('auditorias')}
        >
          Auditorias de Integridad
        </button>
        <button 
          className={vistaActual === 'backups' ? 'active' : ''}
          onClick={() => setVistaActual('backups')}
        >
          Backups de Base de Datos
        </button>
      </div>

      {/* Contenido de Auditorias */}
      {vistaActual === 'auditorias' && (
        <div className="seccion-auditorias">
          <p className="descripcion">
            Reportes semanales que verifican la integridad de los datos del sistema (16 categorias verificadas)
          </p>
          
          {reportes?.auditorias.length === 0 ? (
            <p>No hay auditorias disponibles aun.</p>
          ) : (
            <table>
              <thead>
                <tr>
                  <th>Fecha y Hora</th>
                  <th>Nombre del Archivo</th>
                  <th>Tamaño</th>
                  <th>Acciones</th>
                </tr>
              </thead>
              <tbody>
                {reportes?.auditorias.map((auditoria) => (
                  <tr key={auditoria.nombre}>
                    <td>{formatearFecha(auditoria.fechaCreacion)}</td>
                    <td>{auditoria.nombre}</td>
                    <td>{auditoria.tamanoLegible}</td>
                    <td>
                      <button onClick={() => verAuditoria(auditoria.nombre)}>
                        Ver Reporte
                      </button>
                      <button onClick={() => descargarArchivo('AUDITORIA', auditoria.nombre)}>
                        Descargar
                      </button>
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          )}
        </div>
      )}

      {/* Contenido de Backups */}
      {vistaActual === 'backups' && (
        <div className="seccion-backups">
          <p className="descripcion">
            Respaldos completos de la base de datos generados semanalmente
          </p>
          <div className="advertencia">
            <strong>Atencion:</strong> Los archivos SQL son para uso tecnico. 
            Contactar al administrador para restauracion.
          </div>
          
          {reportes?.backups.length === 0 ? (
            <p>No hay backups disponibles aun.</p>
          ) : (
            <table>
              <thead>
                <tr>
                  <th>Fecha y Hora</th>
                  <th>Nombre del Archivo</th>
                  <th>Tamaño</th>
                  <th>Acciones</th>
                </tr>
              </thead>
              <tbody>
                {reportes?.backups.map((backup) => (
                  <tr key={backup.nombre}>
                    <td>{formatearFecha(backup.fechaCreacion)}</td>
                    <td>{backup.nombre}</td>
                    <td>{backup.tamanoLegible}</td>
                    <td>
                      <button onClick={() => descargarArchivo('BACKUP', backup.nombre)}>
                        Descargar
                      </button>
                    </td>
                  </tr>
                ))}
              </tbody>
            </table>
          )}
        </div>
      )}
    </div>
  );
};

export default ControladoresSistema;

VERSION SIMPLIFICADA (HTML/JS Vanilla):

Si el frontend es mas simple, puedes usar este codigo:

<!DOCTYPE html>
<html>
<head>
  <title>Controladores del Sistema</title>
  <style>
    /* Copiar estilos CSS de arriba */
  </style>
</head>
<body>
  <div id="app"></div>
  
  <script>
    const API_URL = 'http://localhost:8080/api/reportes-sistema';
    
    async function cargarReportes() {
      try {
        const response = await fetch(API_URL);
        const data = await response.json();
        mostrarReportes(data);
      } catch (error) {
        document.getElementById('app').innerHTML = 
          '<p>Error al cargar reportes</p><button onclick="cargarReportes()">Reintentar</button>';
      }
    }
    
    function mostrarReportes(data) {
      let html = '<h1>Controladores del Sistema</h1>';
      
      // Auditorias
      html += '<h2>Auditorias (' + data.totalAuditorias + ')</h2><table>';
      data.auditorias.forEach(a => {
        html += '<tr>';
        html += '<td>' + new Date(a.fechaCreacion).toLocaleString() + '</td>';
        html += '<td>' + a.nombre + '</td>';
        html += '<td>' + a.tamanoLegible + '</td>';
        html += '<td>';
        html += '<button onclick="window.open(\'/api/reportes-sistema/ver/auditoria/' + a.nombre + '\')">Ver</button>';
        html += '<button onclick="window.open(\'/api/reportes-sistema/descargar/AUDITORIA/' + a.nombre + '\')">Descargar</button>';
        html += '</td></tr>';
      });
      html += '</table>';
      
      // Backups
      html += '<h2>Backups (' + data.totalBackups + ')</h2><table>';
      data.backups.forEach(b => {
        html += '<tr>';
        html += '<td>' + new Date(b.fechaCreacion).toLocaleString() + '</td>';
        html += '<td>' + b.nombre + '</td>';
        html += '<td>' + b.tamanoLegible + '</td>';
        html += '<td>';
        html += '<button onclick="window.open(\'/api/reportes-sistema/descargar/BACKUP/' + b.nombre + '\')">Descargar</button>';
        html += '</td></tr>';
      });
      html += '</table>';
      
      document.getElementById('app').innerHTML = html;
    }
    
    cargarReportes();
  </script>
</body>
</html>
